def load_users() -> Dict[str, Any]:
    if os.path.exists(USERS_FILE):
        try:
            with open(USERS_FILE, "r") as f:
                return json.load(f)
        except Exception:
            return {}
    return {}


def save_users(data: Dict[str, Any]) -> None:
    with open(USERS_FILE, "w") as f:
        json.dump(data, f, indent=2)


def register_user(user_id: int) -> int:
    data = load_users()
    data[str(user_id)] = True
    save_users(data)
    return len(data)


def get_user_count() -> int:
    return len(load_users())

# PHONE NUMBER UTILITIES

def extract_numbers_from_text(text: str) -> List[str]:
    """Extract phone number strings from raw text."""
    lines = re.split(r"[\n,;|\t]+", text)
    numbers = []
    for line in lines:
        line = line.strip()
        if not line:
            continue
        # Remove non-digit characters except leading +
        cleaned = re.sub(r"[^\d+]", "", line)
        if cleaned and len(cleaned) >= 7:
            numbers.append(cleaned)
    return numbers


def normalize_number(raw: str) -> Optional[str]:
    """Try to parse and normalize a phone number, returning E.164 format."""
    raw = raw.strip()
    if not raw:
        return None
    # Try with "+" prefix first, then as-is
    for attempt in [raw if raw.startswith("+") else "+" + raw, raw]:
        try:
            parsed = phonenumbers.parse(attempt, None)
            if phonenumbers.is_valid_number(parsed):
                return phonenumbers.format_number(parsed, phonenumbers.PhoneNumberFormat.E164)
        except Exception:
            continue
    # Fallback: just clean it
    cleaned = re.sub(r"\D", "", raw)
    if cleaned:
        return ("+" if not raw.startswith("+") else "") + cleaned
    return None


def get_country_code(number_e164: str) -> str:
    """Return 2-letter country code from E.164 number."""
    try:
        parsed = phonenumbers.parse(number_e164, None)
        region = phonenumbers.region_code_for_number(parsed)
        return region if region else "Unknown"
    except Exception:
        return "Unknown"


def analyze_numbers(raw_numbers: List[str]) -> Dict[str, Any]:
    """Full analysis: valid/invalid, duplicates, country breakdown."""
    valid = []
    invalid = []
    seen = set()
    duplicates = []

    for raw in raw_numbers:
        norm = normalize_number(raw)
        if norm:
            if norm in seen:
                duplicates.append(norm)
            else:
                seen.add(norm)
                valid.append(norm)
        else:
            invalid.append(raw)

    country_breakdown: Dict[str, int] = {}
    for num in valid:
        cc = get_country_code(num)
        country_breakdown[cc] = country_breakdown.get(cc, 0) + 1

    return {
        "total_raw": len(raw_numbers),
        "valid": valid,
        "invalid": invalid,
        "duplicates": duplicates,
        "country_breakdown": country_breakdown,
    }

# FILE PARSING UTILITIES
# =============================================================================

def parse_vcf(content: bytes) -> List[str]:
    """Extract phone numbers from VCF file bytes."""
    text = content.decode("utf-8", errors="replace")
    numbers = []
    for line in text.splitlines():
        line = line.strip()
        if line.upper().startswith("TEL"):
            parts = line.split(":")
            if len(parts) >= 2:
                num = parts[-1].strip()
                numbers.append(num)
    return numbers


def parse_txt(content: bytes) -> List[str]:
    text = content.decode("utf-8", errors="replace")
    return extract_numbers_from_text(text)


def parse_csv_bytes(content: bytes) -> List[str]:
    text = content.decode("utf-8", errors="replace")
    numbers = []
    reader = csv.reader(io.StringIO(text))
    for row in reader:
        for cell in row:
            cleaned = re.sub(r"[^\d+]", "", cell.strip())
            if cleaned and len(cleaned) >= 7:
                numbers.append(cleaned)
    return numbers


def parse_xlsx_bytes(content: bytes) -> List[str]:
    numbers = []
    wb = openpyxl.load_workbook(io.BytesIO(content), read_only=True, data_only=True)
    for ws in wb.worksheets:
        for row in ws.iter_rows(values_only=True):
            for cell in row:
                if cell is not None:
                    val = str(cell).strip()
                    cleaned = re.sub(r"[^\d+]", "", val)
                    if cleaned and len(cleaned) >= 7:
                        numbers.append(cleaned)
    return numbers


def parse_file_bytes(content: bytes, filename: str) -> List[str]:
    """Auto-detect file type and extract numbers."""
    fn = filename.lower()
    if fn.endswith(".vcf"):
        return parse_vcf(content)
    elif fn.endswith(".txt"):
        return parse_txt(content)
    elif fn.endswith(".csv"):
        return parse_csv_bytes(content)
    elif fn.endswith(".xlsx") or fn.endswith(".xls"):
        return parse_xlsx_bytes(content)
    else:
        # Try as text
        return parse_txt(content)


# =============================================================================
# FILE GENERATION UTILITIES
# =============================================================================

def generate_vcf_bytes(contacts: List[Tuple[str, List[str]]]) -> bytes:
    """
    contacts: list of (contact_name, [phone_numbers])
    Returns VCF file bytes.
    """
    lines = []
    for name, phones in contacts:
        lines.append("BEGIN:VCARD")
        lines.append("VERSION:3.0")
        lines.append(f"FN:{name}")
        lines.append(f"N:{name};;;; ")
        for phone in phones:
            lines.append(f"TEL;TYPE=CELL:{phone}")
        lines.append("END:VCARD")
        lines.append("")
    return "\n".join(lines).encode("utf-8")


def generate_txt_bytes(numbers: List[str]) -> bytes:
    return "\n".join(numbers).encode("utf-8")


def generate_csv_bytes(numbers: List[str]) -> bytes:
    output = io.StringIO()
    writer = csv.writer(output)
    writer.writerow(["Phone Number"])
    for num in numbers:
        writer.writerow([num])
    return output.getvalue().encode("utf-8")


def generate_xlsx_bytes(numbers: List[str]) -> bytes:
    wb = Workbook()
    ws = wb.active
    ws.title = "Numbers"
    ws.append(["Phone Number"])
    for num in numbers:
        ws.append([num])
    buf = io.BytesIO()
    wb.save(buf)
    buf.seek(0)
    return buf.read()


def generate_file_bytes(numbers: List[str], fmt: str, contact_name: str = "Contact") -> Tuple[bytes, str]:
    """Generate file in specified format. Returns (bytes, extension)."""
    if fmt == "vcf":
        # One contact per number for simple conversion
        contacts = [(f"{contact_name} {i+1}", [num]) for i, num in enumerate(numbers)]
        return generate_vcf_bytes(contacts), "vcf"
    elif fmt == "txt":
        return generate_txt_bytes(numbers), "txt"
    elif fmt == "excel":
        return generate_xlsx_bytes(numbers), "xlsx"
    elif fmt == "csv":
        return generate_csv_bytes(numbers), "csv"
    else:
        return generate_txt_bytes(numbers), "txt"

def get_ist_now() -> str:
    return datetime.now(IST).strftime("%d %B %Y, %I:%M %p IST")


def format_analysis_report(analysis: Dict[str, Any], source_label: str = "") -> str:
    valid_count = len(analysis["valid"])
    invalid_count = len(analysis["invalid"])
    dup_count = len(analysis["duplicates"])
    total = analysis["total_raw"]

    country_lines = ""
    for country, count in sorted(analysis["country_breakdown"].items(), key=lambda x: -x[1]):
        flag = country_flag(country)
        country_lines += f"  {flag} {country}: {count}\n"

    dup_preview = ""
    if analysis["duplicates"]:
        preview = analysis["duplicates"][:5]
        dup_preview = "\n  " + "\n  ".join(preview)
        if len(analysis["duplicates"]) > 5:
            dup_preview += f"\n  ... and {len(analysis['duplicates']) - 5} more"

    label_part = f"📁 *{source_label}*\n" if source_label else ""

    return (
        f"{label_part}"
        f"📊 *Analysis Report*\n"
        f"{'─' * 30}\n"
        f"📱 Total Numbers: `{total}`\n"
        f"✅ Valid Numbers: `{valid_count}`\n"
        f"❌ Invalid Numbers: `{invalid_count}`\n"
        f"🔁 Duplicate Numbers: `{dup_count}`\n"
        f"\n🌍 *Country Breakdown:*\n{country_lines or '  N/A'}"
        f"{chr(10) + '🔁 *Duplicates:*' + dup_preview if dup_preview else ''}"
    )


def country_flag(code: str) -> str:
    """Return emoji flag for ISO country code."""
    if len(code) != 2 or not code.isalpha():
        return "🌐"
    return chr(ord(code[0].upper()) + 127397) + chr(ord(code[1].upper()) + 127397)


def format_extension(fmt: str) -> str:
    mapping = {"vcf": "vcf", "txt": "txt", "excel": "xlsx", "csv": "csv"}
    return mapping.get(fmt, "txt")


def sanitize_filename(name: str) -> str:
    return re.sub(r"[^\w\- ]", "", name).strip() or "output"

async def fa_receive_file(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:

        file = await doc.get_file()

        await update.message.reply_text(

            reply_markup=back_keyboard(),

async def fa_done(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    files = context.user_data.get("fa_files", [])
    if not files:
        await update.message.reply_text(
            "⚠️ No files received. Please upload at least one file.",
            reply_markup=back_keyboard(),
        )
        return FA_WAITING_FILES

    await update.message.reply_text("🔍 Analyzing files, please wait...")

    for fname, content in files:
        raw_numbers = parse_file_bytes(content, fname)
        analysis = analyze_numbers(raw_numbers)
        report = format_analysis_report(analysis, source_label=fname)
        await update.message.reply_text(report, parse_mode=ParseMode.MARKDOWN)

    context.user_data.pop("fa_files", None)
    await update.message.reply_text(
        "✅ *Analysis complete!*\n\nReturn to main menu:",
        parse_mode=ParseMode.MARKDOWN,
        reply_markup=back_only_keyboard(),
    )
    return ConversationHandler.END

async def fc_choose_format(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    query = update.callback_query
    await query.answer()
    fmt_map = {"fmt_vcf": "vcf", "fmt_txt": "txt", "fmt_excel": "excel", "fmt_csv": "csv"}
    fmt = fmt_map.get(query.data)
    if not fmt:
        return FC_CHOOSE_FORMAT

    files = context.user_data.get("fc_files", [])
    await query.edit_message_text(f"⚙️ Converting {len(files)} file(s) to `{fmt.upper()}`...", parse_mode=ParseMode.MARKDOWN)

    for fname, content in files:
        raw_numbers = parse_file_bytes(content, fname)
        analysis = analyze_numbers(raw_numbers)
        valid_numbers = analysis["valid"]

        base = Path(fname).stem
        out_bytes, ext = generate_file_bytes(valid_numbers, fmt)
        out_name = f"{base}_converted.{ext}"

        bio = io.BytesIO(out_bytes)
        bio.name = out_name
        await query.message.reply_document(document=bio, filename=out_name, caption=f"✅ Converted: `{fname}` → `{out_name}`", parse_mode=ParseMode.MARKDOWN)

    context.user_data.pop("fc_files", None)
    await query.message.reply_text(

async def vm_process(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    files = context.user_data.get("vm_files", [])
    if not files:
        await update.message.reply_text("⚠️ No files uploaded. Please upload at least one file.", reply_markup=back_keyboard())
        return VM_UPLOAD

    vm = context.user_data.get("vm", {})
    per_file = vm.get("per_file", 100)
    contact_name = vm.get("contact_name", "Contact")
    filename_base = vm.get("filename", "Output")
    num_start = vm.get("num_start", 1)
    file_start = vm.get("file_start", 1)
    group_name_base = vm.get("group_name", "Group")

    # Collect all numbers
    all_numbers: List[str] = []
    for fname, content in files:
        raw = parse_file_bytes(content, fname)
        analysis = analyze_numbers(raw)
        all_numbers.extend(analysis["valid"])

    if not all_numbers:
        await update.message.reply_text("⚠️ No valid numbers found in uploaded files.", reply_markup=back_keyboard())
        return ConversationHandler.END

    await update.message.reply_text(f"⚙️ Processing {len(all_numbers)} numbers into VCF files...")

    # Split into chunks
    chunks = [all_numbers[i:i+per_file] for i in range(0, len(all_numbers), per_file)]
    contact_counter = num_start

    for file_idx, chunk in enumerate(chunks):
        file_number = file_start + file_idx
        group_label = f"{group_name_base} {file_number}"
        contacts = []
        for phone in chunk:
            full_name = f"{contact_name} {contact_counter} {group_label}"
            contacts.append((full_name, [phone]))
            contact_counter += 1

        vcf_bytes = generate_vcf_bytes(contacts)
        out_name = f"{filename_base} {file_number}.vcf"
        bio = io.BytesIO(vcf_bytes)
        bio.name = out_name
        await update.message.reply_document(
            document=bio,
            filename=out_name,
            caption=f"📁 `{out_name}` — {len(chunk)} contacts",
            parse_mode=ParseMode.MARKDOWN,
        )

    await update.message.reply_text(
        f"✅ *VCF Maker complete!*\n"
        f"📁 Generated `{len(chunks)}` file(s).\n"
        f"📱 Total contacts: `{len(all_numbers)}`",
        parse_mode=ParseMode.MARKDOWN,
        reply_markup=back_only_keyboard(),
    )
    context.user_data.clear()
    return ConversationHandler.END

async def sv_per_file(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    try:
        per_file = int(update.message.text.strip())
        if per_file < 1:
            raise ValueError
    except ValueError:
        await update.message.reply_text("⚠️ Enter a valid positive number.", reply_markup=back_keyboard())
        return SV_PER_FILE

    files = context.user_data.get("sv_files", [])
    output_name = context.user_data.get("sv_output_name", "Split")
    contact_name_base = context.user_data.get("sv_contact_name", "Contact")

    await update.message.reply_text("⚙️ Splitting files...")

    for orig_fname, content in files:
        ext = Path(orig_fname).suffix.lower().lstrip(".")
        raw = parse_file_bytes(content, orig_fname)
        analysis = analyze_numbers(raw)
        valid = analysis["valid"]

        chunks = [valid[i:i+per_file] for i in range(0, len(valid), per_file)]

        for idx, chunk in enumerate(chunks):
            part_num = idx + 1
            if ext == "vcf":
                contacts = [(f"{contact_name_base} {i+1}", [num]) for i, num in enumerate(chunk)]
                out_bytes = generate_vcf_bytes(contacts)
                out_ext = "vcf"
            elif ext == "csv":
                out_bytes = generate_csv_bytes(chunk)
                out_ext = "csv"
            elif ext in ("xlsx", "xls"):
                out_bytes = generate_xlsx_bytes(chunk)
                out_ext = "xlsx"
            else:
                out_bytes = generate_txt_bytes(chunk)
                out_ext = "txt"

            out_name = f"{output_name} {part_num}.{out_ext}"
            bio = io.BytesIO(out_bytes)
            bio.name = out_name
            await update.message.reply_document(
                document=bio,
                filename=out_name,
                caption=f"✂️ Part {part_num}: `{len(chunk)}` numbers",
                parse_mode=ParseMode.MARKDOWN,
            )

    await update.message.reply_text(
        "✅ *Split complete!*",
        parse_mode=ParseMode.MARKDOWN,
        reply_markup=back_only_keyboard(),
    )
    context.user_data.clear()
    return ConversationHandler.END

async def mf_format_callback(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    query = update.callback_query
    await query.answer()
    fmt_map = {"fmt_vcf": "vcf", "fmt_txt": "txt", "fmt_excel": "excel", "fmt_csv": "csv"}
    fmt = fmt_map.get(query.data)
    if not fmt:
        return MF_FORMAT

    files = context.user_data.get("mf_files", [])
    output_name = context.user_data.get("mf_output_name", "Merged")

    await query.edit_message_text(f"⚙️ Merging {len(files)} file(s)...", parse_mode=ParseMode.MARKDOWN)

    all_numbers: List[str] = []
    for fname, content in files:
        raw = parse_file_bytes(content, fname)
        analysis = analyze_numbers(raw)
        all_numbers.extend(analysis["valid"])

    # De-duplicate
    seen: set = set()
    unique: List[str] = []
    for n in all_numbers:
        if n not in seen:
            seen.add(n)
            unique.append(n)

    out_bytes, ext = generate_file_bytes(unique, fmt)
    out_name = f"{output_name}.{ext}"
    bio = io.BytesIO(out_bytes)
    bio.name = out_name

    await query.message.reply_document(
        document=bio,
        filename=out_name,
        caption=(
            f"✅ *Merge Complete!*\n"
            f"📁 File: `{out_name}`\n"
            f"📱 Total unique numbers: `{len(unique)}`"
        ),
        parse_mode=ParseMode.MARKDOWN,
    )
    await query.message.reply_text("Done!", reply_markup=back_only_keyboard())
    context.user_data.clear()
    return ConversationHandler.END

        parse_mode=ParseMode.MARKDOWN,
        reply_markup=back_keyboard(),
    )

async def fe_numbers(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    new_raw: List[str] = []

    if update.message.document:
        doc = update.message.document
        file = await doc.get_file()
        content = bytes(await file.download_as_bytearray())
        new_raw = parse_file_bytes(content, doc.file_name)
    elif update.message.text:
        new_raw = extract_numbers_from_text(update.message.text)

    analysis = analyze_numbers(new_raw)
    new_numbers = set(analysis["valid"])
    action = context.user_data.get("fe_action", "edit_add")
    files = context.user_data.get("fe_files", [])

    await update.message.reply_text(f"⚙️ Processing {len(files)} file(s)...")

    for fname, content in files:
        ext = Path(fname).suffix.lower().lstrip(".")
        raw = parse_file_bytes(content, fname)
        analysis_orig = analyze_numbers(raw)
        orig_valid = analysis_orig["valid"]

        if action == "edit_add":
            merged = list(dict.fromkeys(orig_valid + list(new_numbers)))  # preserve order, dedupe
            result_numbers = merged
        else:
            result_numbers = [n for n in orig_valid if n not in new_numbers]

        # Generate in same format
        if ext == "vcf":
            contacts = [(f"Contact {i+1}", [num]) for i, num in enumerate(result_numbers)]
            out_bytes = generate_vcf_bytes(contacts)
            out_ext = "vcf"
        elif ext == "csv":
            out_bytes = generate_csv_bytes(result_numbers)
            out_ext = "csv"
        elif ext in ("xlsx", "xls"):
            out_bytes = generate_xlsx_bytes(result_numbers)
            out_ext = "xlsx"
        else:
            out_bytes = generate_txt_bytes(result_numbers)
            out_ext = "txt"

        base = Path(fname).stem
        out_name = f"{base}_edited.{out_ext}"
        bio = io.BytesIO(out_bytes)
        bio.name = out_name
        action_label = "Added" if action == "edit_add" else "Removed"
        await update.message.reply_document(
            document=bio,
            filename=out_name,
            caption=(
                f"✏️ *Edited:* `{fname}`\n"
                f"📱 {action_label} `{len(new_numbers)}` number(s)\n"
                f"📊 Total now: `{len(result_numbers)}`"
            ),
            parse_mode=ParseMode.MARKDOWN,
        )

    await update.message.reply_text("✅ *Editing complete!*", parse_mode=ParseMode.MARKDOWN, reply_markup=back_only_keyboard())
    context.user_data.clear()
    return ConversationHandler.END

async def lm_done(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    screenshots = context.user_data.get("lm_screenshots", [])
    if not screenshots:
        await update.message.reply_text("⚠️ No screenshots received.", reply_markup=back_keyboard())
        return LM_WAITING_SCREENSHOTS

    await update.message.reply_text(f"🔍 Processing {len(screenshots)} screenshot(s) with OCR...")

    all_entries: List[Tuple[str, int]] = []

    for img_bytes in screenshots:
        entries = extract_group_list_from_image(img_bytes)
        all_entries.extend(entries)

    if not all_entries:
        await update.message.reply_text(
            "⚠️ Could not extract any group list data from screenshots.\n\n"
            "Make sure the screenshots contain clear group names and pending request counts.",
            reply_markup=back_only_keyboard(),
        )
        context.user_data.clear()
        return ConversationHandler.END

    # Sort by pending count descending
    all_entries.sort(key=lambda x: -x[1])

    lines = [f"📋 *Group List — {len(all_entries)} entries:*\n"]
    for name, count in all_entries:
        lines.append(f"• {name} — *{count}* pending requests")

    result_text = "\n".join(lines)

    # Send as message if short, else as document
    if len(result_text) < 4000:
        await update.message.reply_text(result_text, parse_mode=ParseMode.MARKDOWN, reply_markup=back_only_keyboard())
    else:
        bio = io.BytesIO(result_text.encode("utf-8"))
        bio.name = "group_list.txt"
        await update.message.reply_document(document=bio, filename="group_list.txt", caption="📋 Group list extracted via OCR.")
        await update.message.reply_text("✅ Done!", reply_markup=back_only_keyboard())

    context.user_data.clear()
    return ConversationHandler.END


def extract_group_list_from_image(img_bytes: bytes) -> List[Tuple[str, int]]:
    """Use OCR to extract group name and pending counts from an image."""
    if not OCR_AVAILABLE:
        return []

    try:
        image = Image.open(io.BytesIO(img_bytes))
        text = pytesseract.image_to_string(image)
        return parse_group_list_text(text)
    except Exception as e:
        logger.error(f"OCR error: {e}")
        return []


def parse_group_list_text(text: str) -> List[Tuple[str, int]]:
    """Parse OCR text to find group names and pending request counts."""
    entries = []
    lines = text.splitlines()

    # Patterns to detect "X pending" or "pending: X" or "X requests"
    pending_patterns = [
        re.compile(r"(.+?)\s*[-–:]\s*(\d+)\s*pending", re.IGNORECASE),
        re.compile(r"(.+?)\s*[-–:]\s*pending\s*(\d+)", re.IGNORECASE),
        re.compile(r"(.+?)\s*[-–:]\s*(\d+)\s*requests?", re.IGNORECASE),
        re.compile(r"(\d+)\s+(.+?)\s+pending", re.IGNORECASE),
    ]

    for line in lines:
        line = line.strip()
        if not line:
            continue
        for pattern in pending_patterns:
            m = pattern.match(line)
            if m:
                groups = m.groups()
                try:
                    if groups[0].isdigit():
                        count = int(groups[0])
                        name = groups[1].strip()
                    else:
                        name = groups[0].strip()
                        count = int(groups[1])
                    if name and count > 0:
                        entries.append((name, count))
                        break
                except (ValueError, IndexError):
                    continue

    # Fallback: look for lines with numbers that might be counts
    if not entries:
        for i, line in enumerate(lines):
            nums = re.findall(r"\b(\d+)\b", line)
            if nums and i > 0:
                name_line = lines[i - 1].strip() if i > 0 else line
                if name_line and not name_line.isdigit():
                    try:
                        entries.append((name_line, int(nums[0])))
                    except ValueError:
                        pass

    return entries

async def rf_new_name(update: Update, context: ContextTypes.DEFAULT_TYPE) -> int:
    new_name = update.message.text.strip()
    rf = context.user_data.get("rf", {})
    orig_fname = rf.get("filename", "file")
    content = rf.get("content", b"")
    action = rf.get("action", "rename_file")
    ext = Path(orig_fname).suffix.lower()

    if action == "rename_file":
        new_name_clean = sanitize_filename(new_name)
        out_name = f"{new_name_clean}{ext}"
        bio = io.BytesIO(content)
        bio.name = out_name
        await update.message.reply_document(
            document=bio,
            filename=out_name,
            caption=f"📝 Renamed: `{orig_fname}` → `{out_name}`",
            parse_mode=ParseMode.MARKDOWN,
        )

    else:  # rename_contacts
        raw = parse_file_bytes(content, orig_fname)
        analysis = analyze_numbers(raw)
        valid = analysis["valid"]

        ext_clean = ext.lstrip(".")
        if ext_clean == "vcf":
            contacts = [(f"{new_name} {i+1}", [num]) for i, num in enumerate(valid)]
            out_bytes = generate_vcf_bytes(contacts)
            out_ext = "vcf"
        elif ext_clean == "csv":
            out_bytes = generate_csv_bytes(valid)
            out_ext = "csv"
        elif ext_clean in ("xlsx", "xls"):
            out_bytes = generate_xlsx_bytes(valid)
            out_ext = "xlsx"
        else:
            out_bytes = generate_txt_bytes(valid)
            out_ext = "txt"

        out_name = f"{Path(orig_fname).stem}_renamed.{out_ext}"
        bio = io.BytesIO(out_bytes)
        bio.name = out_name
        await update.message.reply_document(
            document=bio,
            filename=out_name,
            caption=(
                f"✏️ Contacts renamed to `{new_name} N`\n"
                f"📱 Total: `{len(valid)}`"
            ),
            parse_mode=ParseMode.MARKDOWN,
        )

    await update.message.reply_text("✅ *Done!*", parse_mode=ParseMode.MARKDOWN, reply_markup=back_only_keyboard())
    context.user_data.clear()
    return ConversationHandler.END
